from datetime import datetime # Anniversary date (year, month, day) anniversary_date = datetime(2023, 10, 15) # Today's date today = datetime.now() # Calculate the difference in days days_since_anniversary = (today - anniversary_date).days # Print the message if days_since_anniversary == 0: print("Happy Anniversary!") elif days_since_anniversary > 0: print(f"It has been {days_since_anniversary} days since the anniversary.") else: print(f"The anniversary is in {-days_since_anniversary} days!")
